Dim ImDone As Boolean
Sub GTST3()
ActivePresentation.SlideShowWindow.View.GotoSlide (3)
Timer1
End Sub
Public Sub CommandButton1_Click()

    'Checks for empty submission
    If A2 = vbNullString Then
        MsgBox "Please enter an answer"
    Exit Sub
    End If
    
    'Sets the timer to True so that it will exit the timer when done.
    ImDone = True
    
    'Checks answer
    If A2 = ActivePresentation.Slides(3).Shapes("Answer2").TextFrame.TextRange.Text Then
        MsgBox "Correct!  +200 Points!"
    Call Slide1.CorrectT
    Call Slide27.CorrectT
    Else
        MsgBox "Wrong.  Correct Answer: " & ActivePresentation.Slides(3).Shapes("Answer2").TextFrame.TextRange.Text & ". -200 Points."
    Call Slide1.IncorrectT
    Call Slide27.IncorrectT
    End If
    
    'Send the presentation to slide 1
    ActivePresentation.SlideShowWindow.View.GotoSlide (1)
        
    'Hide square to asnwered question
    ap.Slides(1).Shapes("c1r2").visibility = msoFalse
    ap.SlideShowWindow.View.First
    
    'Checks for end game
    Call AllQuestionsAnswered

End Sub
Sub Timer1()
    time = Now()
    count = 20
    time = DateAdd("s", count, time)

    Do Until time < Now()
        DoEvents
        ActivePresentation.Slides(3).Shapes("Timer1").TextFrame.TextRange.Text = Format((time - Now()), "ss")
    Loop
    If ImDone = True Then
    Exit Sub
    Else
        MsgBox "You ran out of time!  Correct Answer: " & ActivePresentation.Slides(3).Shapes("Answer2").TextFrame.TextRange.Text & "."
     End If
End Sub
