Sub StartGame()
Dim oOutlook As Object
Dim sld As Slide
Dim L As Long

Set ap = ActivePresentation
    
    'Makes sure Outlook is open
    On Error Resume Next
    Set oOutlook = GetObject(, "Outlook.Application")
    On Error GoTo 0

    If oOutlook Is Nothing Then
        Set oOutlook = CreateObject("Outlook.Application")
        MsgBox "You will need to open Outlook before you may begin!"
    Exit Sub
    End If
    
    'Hides Covers
 On Error Resume Next
 For Each sld In ActivePresentation.Slides
 For L = sld.Shapes.count To 1 Step -1
 If sld.Shapes(L).Name = ("Delete_Cover") Then sld.Shapes(L).Visible = msoFalse
 Next L
 Next sld
End Sub

Sub CoverFinalProduct()
'Makes covers visible
 On Error Resume Next
 For Each sld In ActivePresentation.Slides
 For L = sld.Shapes.count To 1 Step -1
 If sld.Shapes(L).Name = ("Delete_Cover") Then sld.Shapes(L).Visible = msoTrue
 Next L
 Next sld
  
 On Error Resume Next
 For Each sld In ActivePresentation.Slides
 For L = sld.Shapes.count To 1 Step -1
 If sld.Shapes(L).Name = ("CommandButton1") Then sld.Shapes(L).Visible = msoFalse
 Next L
 Next sld
 ActivePresentation.Slides(1).Shapes("Score").Visible = msoFalse
 ActivePresentation.Slides(27).Shapes("Score").Visible = msoFalse
End Sub
